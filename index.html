        code:--

        document
        .getElementById("ss1")
        .addEventListener("mouseover", function () {
            gsap
            .to("#ss1", { delay: 0, x: 350, y: 250, scaleX: 2, scaleY: 2 });
          });

          document
          .getElementById("ss1")
          .addEventListener("mouseout", function () {
            gsap
            .to("#ss1", { delay: 0, x: 0, y: 0, scaleX: 1, scaleY: 1 });
          });

        --------------------------------------------
         html:-   
         
         <....... onmouseover="imageEnlarge(event,1,2,3,4,5,6)"/>
                                            -----
         js:-

          function imageEnlarge(e, ...args) {  
                                -----
               event.preventDefault();
               console.log(e.target.id);            //ss1 or ss2
               console.log(args);                   //[1, 2, 3, 4, 5, 6]
            }
